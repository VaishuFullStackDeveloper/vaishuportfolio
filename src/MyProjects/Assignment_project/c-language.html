Code (Terminal Input)

#include<stdio.h> int main() { int year; printf("Enter a year:"); scanf("%d", &year); (year % 4 == 0 && year % 100 != 0) || (year % 400 == 0) ? printf("%d is a leap year.\n", year) : printf("%d is not a leap year", year); return 0; }

Input

Enter a year: 2024

Output

2024 is a leap year.

Code (Factorial)

#include<stdio.h> int main() { int num, i, factorial = 1; printf("Enter a number: "); scanf("%d", &num); for(i = 1; i <= num; i++) { factorial *= i; } printf("Factorial of %d is %d\n", num, factorial); return 0; }

Input

Enter a number: 5

Output

Factorial of 5 is 120.

Code (Prime Number Check)

#include<stdio.h> int main() { int num, i, flag = 0; printf("Enter a number: "); scanf("%d", &num); for(i = 2; i <= num/2; i++) { if(num % i == 0) { flag = 1; break; } } if(flag == 0) printf("%d is a prime number.\n", num); else printf("%d is not a prime number.\n", num); return 0; }

Input

Enter a number: 7

Output

7 is a prime number.
